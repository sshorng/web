/** 國文心智圖引導助理 — 旗艦升級後端 */ const SHEET_NAME = '課文資料庫'; const SETTINGS_SHEET = '系統設定'; const HEADERS = ['課次', '課名', '講義內容', '分析模式', '參考心智圖', '顯示順序', '建立時間', '更新時間']; /** * 初始化工作表 * 執行此函式可建立所需的工作表與欄位 */ function initializeSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); // 建立課文資料庫 if (!ss.getSheetByName(SHEET_NAME)) { const s = ss.insertSheet(SHEET_NAME); s.getRange(1, 1, 1, HEADERS.length).setValues([HEADERS]).setFontWeight('bold'); s.setFrozenRows(1); s.setColumnWidth(1, 100); s.setColumnWidth(2, 150); s.setColumnWidth(3, 400); } // 建立系統設定分頁 if (!ss.getSheetByName(SETTINGS_SHEET)) { const s = ss.insertSheet(SETTINGS_SHEET); s.getRange(1, 1, 1, 2).setValues([['設定項', '內容']]).setFontWeight('bold'); s.appendRow(['Gemini_API_Keys', '請在此填入 API Keys (一行一個)']); s.appendRow(['Gemini_Model', 'gemini-1.5-pro']); s.appendRow(['Gemini_Model_Student', 'gemini-2.0-flash']); s.appendRow(['Teacher_Password', 'admin']); s.setColumnWidth(2, 500); } } /** 取得特定設定值 */ function getSetting(key) { const s = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SETTINGS_SHEET); if (!s) return null; const data = s.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] === key) return data[i][1]; } return null; } /** 取得所有 API Key 並打亂順序以平衡負載 */ function getAllApiKeys() { const keysStr = getSetting('Gemini_API_Keys') || ''; // 支援換行分隔，過濾掉預設說明文字 let keys = keysStr.split(/\r?\n/).map(k => k.trim()).filter(k => k && !k.includes('請在此填入')); if (keys.length === 0) throw new Error('未設定 API Key，請在試算表「系統設定」分頁中填入。'); // 原地打亂陣列 (Fisher-Yates Shuffle) for (let i = keys.length - 1; i > 0; i--) { const j = Math.floor(Math.random() * (i + 1)); [keys[i], keys[j]] = [keys[j], keys[i]]; } return keys; } /** 通用回傳格式 */ function res(obj) { return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON); } /** GET 請求門口 */ function doGet(e) { const action = (e.parameter && e.parameter.action) || ''; try { if (action === 'list') return res(listLessons()); if (action === 'get') return res(getLesson(e.parameter.id)); return res({ success: false, error: '未知動作' }); } catch (err) { return res({ success: false, error: err.message }); } } /** POST 請求門口 */ function doPost(e) { let body; try { body = JSON.parse(e.postData.contents); } catch (err) { return res({ success: false, error: '無效的 JSON 資料' }); } try { // 1. 代理 Gemini 請求 (核心功能：支援多 Key 自動輪詢與錯誤重試) if (body.action === 'proxy_gemini') { const keys = getAllApiKeys(); const source = body.source || 'teacher'; let model = ''; if (source === 'student') { model = getSetting('Gemini_Model_Student') || getSetting('Gemini_Model') || 'gemini-2.0-flash'; } else { model = getSetting('Gemini_Model') || 'gemini-1.5-pro'; } let lastErrorResponse = null; // 循序嘗試所有金鑰 for (let i = 0; i < keys.length; i++) { const apiKey = keys[i]; const url = 'https://generativelanguage.googleapis.com/v1beta/models/' + model + ':generateContent?key=' + apiKey; const options = { method: 'post', contentType: 'application/json', payload: JSON.stringify(body.payload), muteHttpExceptions: true }; try { const response = UrlFetchApp.fetch(url, options); const statusCode = response.getResponseCode(); // 如果成功 (200)，直接回傳 if (statusCode === 200) { return ContentService.createTextOutput(response.getContentText()).setMimeType(ContentService.MimeType.JSON); } // 紀錄失敗原因並繼續嘗試下一個 Key console.warn('API Key 第 ' + (i + 1) + ' 組失敗 (狀態碼 ' + statusCode + ')，轉換中...'); lastErrorResponse = response; } catch (e) { console.error('API 請求異常 (Key 第 ' + (i + 1) + ' 組): ' + e.message); } } // 如果所有 Key 都失敗了，回傳最後一個錯誤訊息 if (lastErrorResponse) { return ContentService.createTextOutput(lastErrorResponse.getContentText()).setMimeType(ContentService.MimeType.JSON); } else { return res({ success: false, error: '所有 API Key 均嘗試失敗或無效。' }); } } // 2. 教師權限驗證 if (body.action === 'auth') { const correctPw = getSetting('Teacher_Password') || 'admin'; return res({ success: body.password === correctPw }); } // 2. 獲取系統設定 if (body.action === 'get_settings') { return res({ success: true, settings: { Gemini_API_Keys: getSetting('Gemini_API_Keys'), Gemini_Model: getSetting('Gemini_Model'), Gemini_Model_Student: getSetting('Gemini_Model_Student') } }); } // 3. 儲存系統設定 if (body.action === 'save_settings') { const s = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SETTINGS_SHEET); const data = s.getDataRange().getValues(); const updates = body.settings; for (let k in updates) { let found = false; for (let i = 1; i < data.length; i++) { if (data[i][0] === k) { s.getRange(i + 1, 2).setValue(updates[k]); found = true; break; } } if (!found) s.appendRow([k, updates[k]]); } return res({ success: true }); } // 3. 儲存課文 if (body.action === 'save') return res(saveLesson(body.lesson)); // 4. 刪除課文 if (body.action === 'delete') return res(deleteLesson(body.id)); return res({ success: false, error: '未知動作' }); } catch (err) { return res({ success: false, error: err.message }); } } /** 取得資料表物件 */ function getSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); return ss.getSheetByName(SHEET_NAME) || (initializeSheet() || ss.getSheetByName(SHEET_NAME)); } /** 列出所有課文 */ function listLessons() { const s = getSheet(); const lr = s.getLastRow(); if (lr < 2) return { success: true, lessons: [] }; const d = s.getRange(2, 1, lr - 1, HEADERS.length).getValues(); const lessons = d.map(r => ({ id: r[0], // 課次 title: r[1], // 課名 mode: r[3], // 分析模式 hasTargetMap: !!r[4], // 是否有心智圖 order: r[5] || 0, createdAt: r[6] })); lessons.sort((a, b) => (a.order || 0) - (b.order || 0)); return { success: true, lessons }; } /** 取得特定課文詳情 (支援合併多行內容) */ function getLesson(id) { const s = getSheet(); const lr = s.getLastRow(); if (lr < 2) return { success: false, error: '資料庫為空' }; const d = s.getRange(2, 1, lr - 1, HEADERS.length).getValues(); let lesson = null; let combinedHandout = []; for (let i = 0; i < d.length; i++) { if (d[i][0].toString() === id.toString()) { if (!lesson) { let tm = null; try { tm = JSON.parse(d[i][4]); } catch (e) { } lesson = { id: d[i][0], title: d[i][1], mode: d[i][3], targetMap: tm, order: d[i][5] }; } // 收集所有相同 ID 的講義內容 if (d[i][2]) combinedHandout.push(d[i][2].toString()); } } if (lesson) { lesson.handout = combinedHandout.join('\n'); // 以換行符號連接所有行 return { success: true, lesson: lesson }; } return { success: false, error: '查無此課次：' + id }; } /** 儲存或新增課文 */ function saveLesson(lesson) { if (!lesson || !lesson.id || !lesson.title) return { success: false, error: '課次與課名不可為空' }; const s = getSheet(); const lr = s.getLastRow(); const now = new Date().toLocaleString('zh-TW'); // 檢查是否已存在 (以課次 ID 為準) if (lr >= 2) { const ids = s.getRange(2, 1, lr - 1, 1).getValues(); for (let i = 0; i < ids.length; i++) { // 轉為字串比較以免 ID 為純數字時失敗 if (ids[i][0].toString() === lesson.id.toString()) { const r = i + 2; const updateData = [ lesson.id, lesson.title, lesson.handout || '', lesson.mode || 'auto', lesson.targetMap ? JSON.stringify(lesson.targetMap) : '', lesson.order || 0 ]; s.getRange(r, 1, 1, 6).setValues([updateData]); s.getRange(r, 8).setValue(now); return { success: true, id: lesson.id }; } } } // 若不存在則新增一行 s.appendRow([ lesson.id, lesson.title, lesson.handout || '', lesson.mode || 'auto', lesson.targetMap ? JSON.stringify(lesson.targetMap) : '', lesson.order || lr, now, now ]); return { success: true, id: lesson.id }; } /** 刪除課文 */ function deleteLesson(id) { if (!id) return { success: false, error: '缺少課次 ID' }; const s = getSheet(); const lr = s.getLastRow(); if (lr < 2) return { success: false, error: '資料庫為空' }; const ids = s.getRange(2, 1, lr - 1, 1).getValues(); for (let i = 0; i < ids.length; i++) { if (ids[i][0].toString() === id.toString()) { s.deleteRow(i + 2); return { success: true }; } } return { success: false, error: '刪除失敗：不存在的課次' }; }
